    def Name.new(namespace_prefix, namespace_uri, local_name)
      key = [namespace_prefix, namespace_uri, local_name, self]
      NameCache.fetch(key) {
        0.upto(2) {|i| key[i] = key[i].dup.freeze if key[i] }
        NameCache[key] = super(key[0], key[1], key[2])
      }
    end
